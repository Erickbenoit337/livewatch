FROM python:3.11-slim

WORKDIR /app

# Dépendances système (ffmpeg pour l'enregistrement, curl pour healthcheck)
RUN apt-get update && apt-get install -y --no-install-recommends \
    ffmpeg \
    curl \
    libpq-dev \
    gcc \
    && rm -rf /var/lib/apt/lists/*

# Dépendances Python
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Code source
COPY Livewatch.py .
COPY .env.example .env

# Répertoires runtime
RUN mkdir -p static templates static/uploads static/thumbnails static/recordings

EXPOSE 8000
CMD ["python3", "Livewatch.py"]
